"""
sft_luxun.py - 对预训练模型进行有监督微调 (SFT)
基于预训练的鲁迅 Transformer，使用多轮对话数据训练助手行为。
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import math
import os
import json
from tokenizers import Tokenizer

# ---- 对话数据格式化 ----
def format_conversation(conv):
    """
    将对话列表格式化为模型输入字符串。
    例如: <|user|>你好<|assistant|>你好！<|user|>再见<|assistant|>再见！
    """
    special_tokens = {
        "system": "<|system|>",
        "user": "<|user|>",
        "assistant": "<|assistant|>",
        "tool": "<|tool|>",
    }
    text = ""
    for turn in conv:
        role = turn.get("role", "user")
        content = turn.get("content", "")
        if role in special_tokens:
            text += special_tokens[role] + content
    return text

# ---- 加载数据 ----
def load_sft_data(filepath):
    """加载 SFT 数据 (jsonl)"""
    data = []
    with open(filepath, 'r', encoding='utf-8') as f:
        for line in f:
            sample = json.loads(line)
            conversations = sample["conversations"]
            data.append(conversations)
    return data

# ---- SFT 数据集 ----
class SFTDataset(torch.utils.data.Dataset):
    """SFT 数据集，仅对 assistant 回复部分计算损失"""
    def __init__(self, data, tokenizer, max_len=256):
        self.samples = []
        bos_id = tokenizer.token_to_id("<|bos|>")
        eos_id = tokenizer.token_to_id("<|eos|>")
        pad_id = tokenizer.token_to_id("<|pad|>")
        user_id = tokenizer.token_to_id("<|user|>")
        assistant_id = tokenizer.token_to_id("<|assistant|>")

        for conv in data:
            text = format_conversation(conv)
            ids = tokenizer.encode(text).ids
            # labels：-100 表示忽略，只保留 assistant 部分
            labels = [-100] * len(ids)
            i = 0
            while i < len(ids):
                if ids[i] == assistant_id:
                    j = i + 1
                    while j < len(ids) and ids[j] not in [user_id, assistant_id, pad_id, bos_id]:
                        j += 1
                    for k in range(i, j):
                        labels[k] = ids[k]
                    i = j
                else:
                    i += 1
            # 截断或填充
            if len(ids) > max_len:
                ids = ids[:max_len]
                labels = labels[:max_len]
            else:
                padding = [pad_id] * (max_len - len(ids))
                ids = ids + padding
                labels = labels + [-100] * (max_len - len(labels))
            self.samples.append((ids, labels))

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, idx):
        ids, labels = self.samples[idx]
        return torch.tensor(ids, dtype=torch.long), torch.tensor(labels, dtype=torch.long)

# ---- 训练主循环 ----
def train_sft():
    batch_size, epochs, lr = 4, 5, 1e-5
    max_len = 256
    device = 'cuda' if torch.cuda.is_available() else 'cpu'

    # 加载分词器
    tokenizer = Tokenizer.from_file("tokenizer_luxun/tokenizer.json")
    vocab_size = len(tokenizer.get_vocab())

    # 加载预训练模型
    ckpt = torch.load("checkpoints_luxun2/best.pth", map_location=device)
    model = FullTransformer(vocab_size, d_model=64,
                            n_layers=3, n_heads=4).to(device)
    model.load_state_dict(ckpt)

    # 冻结参考模型
    ref_model = FullTransformer(vocab_size, d_model=64,
                                n_layers=3, n_heads=4).to(device)
    ref_model.load_state_dict(model.state_dict())
    for param in ref_model.parameters():
        param.requires_grad = False

    # 构建数据加载器
    dataset = SFTDataset(load_sft_data("sft_data.jsonl"),
                          tokenizer, max_len)
    loader = torch.utils.data.DataLoader(dataset, batch_size=batch_size,
                                         shuffle=True)

    optimizer = torch.optim.AdamW(model.parameters(), lr=lr)
    scaler = torch.amp.GradScaler('cuda', enabled=(device != 'cpu'))

    model.train()
    for epoch in range(epochs):
        total_loss = 0
        for x, y in loader:
            x, y = x.to(device), y.to(device)
            with torch.amp.autocast('cuda', dtype=torch.bfloat16):
                logits = model(x)
                loss = F.cross_entropy(
                    logits.view(-1, vocab_size),
                    y.view(-1),
                    ignore_index=-100
                )
            scaler.scale(loss).backward()
            scaler.step(optimizer)
            scaler.update()
            optimizer.zero_grad()
            total_loss += loss.item()
        print(f"Epoch {epoch+1}: loss = {total_loss/len(loader):.4f}")

    # 保存模型
    os.makedirs("checkpoints_sft", exist_ok=True)
    torch.save(model.state_dict(), "checkpoints_sft/sft_model.pth")
    torch.save(ref_model.state_dict(), "checkpoints_sft/reference_model.pth")
    print("SFT 训练完成！")
SFT：有监督微调
一句话总结：把"会读书的模型"变成"会聊天的模型"——用标准对话手把手教它怎么好好说话。

0 SFT 是什么？

预训练模型就像一个饱读诗书但说话阴阳怪气的鲁迅——读了整个互联网，满腹经纶，但你问它"你好"，它可能回你"今天天气真好"或者直接开骂。

SFT（Supervised Fine-Tuning）就是给这位才子请了一位一对一私教，用精心标注的「问-答」对，教它"人家问你好，你得说'你好！'"。

🍜 比喻：预训练 = 孩子在图书馆自学了 10 年；SFT = 家长拿着标准答案，一句一句教孩子"这道题该这么答"。孩子还是那个孩子，只是学会了应试技巧。
核心公式
$$ \mathcal{L}_{\text{SFT}} = -\sum_{t \in \text{assistant}} \log P_\theta(y_t \mid y_{<t}, x) $$

只在 assistant 回复的 token 上计算交叉熵损失，user 部分全部屏蔽。

1 整体架构：训练流程一览

整个 SFT 流程可以用一张图说清楚：

原始对话数据 JSONL 格式，每行一轮对话 │ ▼ format_conversation() ← 加上特殊 token，拼成一长串 │ ▼ Tokenizer.encode() ← 文本 → token id 序列 │ ▼ SFTDataset ← 核心：构造 labels，-100 屏蔽非 assistant │ ▼ DataLoader ← batch=4, shuffle=True │ ▼ 训练循环 × 5 epochs ← forward → cross_entropy(ignore_index=-100) → backward │ ▼ 保存两个模型 ← sft_model.pth + reference_model.pth
💡 关键设计：SFT 同时保存两个模型副本——训练后的 sft_model 用于后续微调（PPO 的 Actor / DPO 的 Policy），冻结的 reference_model 作为"原来的样子"，提供 KL 约束基准。

2 第一步：对话数据格式化

format_conversation() 函数详解

这个函数做的事情很简单：把一堆零散的对话轮次，拼成一个带特殊 token 标记的长字符串。

🍳 比喻：想象你在整理会议记录。每个人的发言都贴上"张三："、"李四："的标签，然后按时间顺序连成一整段。format_conversation 做的就是这个——只不过"张三"变成了 <|user|>，"李四"变成了 <|assistant|>。

输入的对话数据长这样：

[ {"role": "user", "content": "你是谁？"}, {"role": "assistant", "content": "我是鲁迅，周树人也。"}, {"role": "user", "content": "写首诗吧。"}, {"role": "assistant", "content": "横眉冷对千夫指..."} ]

输出变成这样：

<|user|>你是谁？<|assistant|>我是鲁迅，周树人也。<|user|>写首诗吧。<|assistant|>横眉冷对千夫指...

代码的核心逻辑：

  • 用一个字典把角色名映射到特殊 token（<|user|>、<|assistant|> 等）
  • 遍历每一轮对话，把 special_token + content 拼接起来
  • 最终返回一个完整的文本串，准备喂给 Tokenizer
🔑 为什么要加特殊 token？
模型需要知道"谁在说话"。特殊 token 就像对话的路标——模型看到 <|assistant|> 就知道："该我表演了！" 后续 SFTDataset 也依赖这些 token 来定位 assistant 的回复片段。

3 第二步：SFTDataset —— 标签的魔法

这是整个 SFT 中最精妙的部分。为什么说它精妙？因为这里藏着一个灵魂拷问：

🤔 灵魂拷问：对话有 user 和 assistant 两部分，模型对整段话的每个位置都在预测下一个 token。那——损失应该算在谁头上？

答案是：只算 assistant 的！ 因为我们的目标是教模型"怎么回答"，不是教它"怎么提问"。

数据流详解

看一段经过 Tokenizer 编码后的 token id 序列：

ids: [bos, user_id, 你, 好, ?, assistant_id, 你, 好, !, user_id, 再, 见, assistant_id, 再, 见, !, eos, pad, pad, ...] labels: [-100, -100, -100,-100,-100, -100, 你, 好, !, -100, -100,-100, -100, 再, 见, !, -100,-100,-100,...]

关键点：

  • labels 初始全部设为 -100（PyTorch 的 ignore_index，损失函数会跳过）
  • 扫描 ids，找到 assistant_id 的位置
  • 从 assistant_id 开始，一直扫到下一个 user_id 或 pad，把这一段的 labels 设为对应的 ids 值
  • 最后做截断/填充到 max_len=256
user 部分 labels = -100 → assistant 部分 labels = ids → 计算损失 ✓
损失计算
$$ \mathcal{L} = -\frac{1}{N} \sum_{t=1}^{T} \mathbb{1}[y_t \neq -100] \cdot \log P_\theta(y_t \mid y_{<t}, x) $$

当 labels[t] == -100 时，cross_entropy 的 ignore_index=-100 会直接跳过该位置，不参与梯度计算。

🎯 比喻：这就像是考试批改——老师只批改"学生作答"的部分，不批改"题目本身"。user 的输入是"题目"，assistant 的回复才是"答案"，我们只关心答案写得对不对。
🔑 为什么 include assistant_id 本身？
注意代码中 for k in range(i, j)，其中 i 是 assistant_id 的位置。这意味着 <|assistant|> 这个 token 也参与了损失计算——模型不仅学"怎么回复"，还学"什么时候该开始回复"。

4 第三步：训练主循环

train_sft() 函数是整个流程的"总指挥"，负责四件事：加载模型 → 冻结参考模型 → 训练 → 保存。

4.1 加载预训练模型

ckpt = torch.load("checkpoints_luxun2/best.pth", map_location=device) model = FullTransformer(vocab_size, d_model=64, n_layers=3, n_heads=4) model.load_state_dict(ckpt)

这里加载的是预训练阶段的 checkpoint——模型已经学会了语言的基本结构和知识，但还不会"好好聊天"。

📖 比喻：预训练模型 = 读了 10 年书的学生，知识储备丰富但没经过答题训练。SFT = 拿着标准答案来教他"这道题该这么答"。

4.2 冻结参考模型（Reference Model）

ref_model = FullTransformer(...) # 和 model 结构完全一样 ref_model.load_state_dict(model.state_dict()) # 复制一份初始权重 for param in ref_model.parameters(): param.requires_grad = False # 冻结！不参与训练

这段代码做了一件看起来"多余"的事：复制一份模型，然后冻住它。为什么要这么做？

💡 为什么需要 Reference Model？
  • 在 PPO（强化学习对齐）中：ref_model 提供 KL 散度的基准，防止 SFT 模型"跑偏太远"
  • 在 DPO（直接偏好优化）中：ref_model 提供概率比的分母，让模型学会"偏好哪种回答"
  • 简单说：ref_model 是训练前的"照片"，用来对比训练后的变化，防止模型"忘本"
📸 比喻：ref_model 就像减肥前拍的"Before 照片"。减肥（训练）过程中，你一直在对比："我现在比 Before 照片瘦了多少？有没有走极端？"没有这张照片，你就不知道自己跑偏了多远。

4.3 训练循环

model.train() for epoch in range(epochs): # 5 个 epoch for x, y in loader: # batch_size=4 with torch.amp.autocast(): # 混合精度训练，省显存 logits = model(x) # 前向传播 loss = F.cross_entropy( # 核心损失函数 logits.view(-1, vocab_size), y.view(-1), ignore_index=-100 # 跳过 -100 的位置 ) scaler.scale(loss).backward() # 反向传播 scaler.step(optimizer) # 更新参数 scaler.update() optimizer.zero_grad()

训练循环的几个关键细节：

  • torch.amp.autocast()：混合精度训练（bfloat16），在不损失精度的前提下减少显存占用和加速计算
  • ignore_index=-100：这是整个 SFT 的"灵魂"——labels 中值为 -100 的位置被完全忽略，只在 assistant 回复上计算损失
  • GradScaler：配合 autocast 使用，防止梯度下溢
  • lr=1e-5：学习率很小！因为我们在"微调"预训练模型，步子太大容易"忘掉"已学知识
超参数设置
batch_size 4 小 batch，适合小显存
epochs 5 多轮训练，充分学习对话格式
lr 1e-5 微调专用小学习率
max_len 256 序列最大长度

4.4 保存模型

torch.save(model.state_dict(), "checkpoints_sft/sft_model.pth") torch.save(ref_model.state_dict(), "checkpoints_sft/reference_model.pth")

保存两个模型，各有用途：

sft_model.pth → PPO 的 Actor / DPO 的 Policy
reference_model.pth → KL 惩罚基准 / 概率比参考

5 关键要点回顾

🔑 要点一：特殊 token 是对话的骨架
<|user|>、<|assistant|> 等特殊 token 不仅用于格式化输入，还用于定位 assistant 回复片段，是标签构造的核心依据。
🔑 要点二：-100 是损失的"隐身衣"
PyTorch 的 cross_entropy 会跳过 ignore_index=-100 的位置。我们利用这个特性，把所有非 assistant 的 token 标为 -100，实现"只学回答，不学提问"。
🔑 要点三：Reference Model 是未来的锚点
冻结的 ref_model 不参与训练，但它会在后续的 PPO/DPO 中作为"原始版本"提供约束——防止模型在对齐过程中"跑偏太远"或"忘记预训练知识"。
🔑 要点四：小学习率是微调的铁律
lr=1e-5 远小于预训练的学习率。模型已经在预训练中学到了大量知识，SFT 只需要"微调"它的行为模式，步子太大会导致灾难性遗忘（Catastrophic Forgetting）。
🚀 SFT 之后是什么？
SFT 让模型学会了"像助手一样回答"，但还只是模仿，无法超越人类标注质量。要让模型真正"对齐人类偏好"，还需要后续的 PPO（强化学习）或 DPO（直接偏好优化）——这就是 RLHF 的后续步骤。
